    Session คืออะไร?¶

    Session คือวิธีเก็บข้อมูลผู้ใช้ชั่วคราวบน Server

    ┌─────────────────────────────────────────────────────────────────────┐
│                         Session ทำงานอย่างไร                        │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  1. Login สำเร็จ → Server สร้าง Session ID                          │
│                                                                     │
│  2. Server ส่ง Session ID กลับใน Cookie                             │
│     Set-Cookie: PHPSESSID=abc123...                                │
│                                                                     │
│  3. Browser เก็บ Cookie ไว้                                         │
│                                                                     │
│  4. Request ต่อไป Browser ส่ง Cookie ไปด้วยทุกครั้ง                   │
│     Cookie: PHPSESSID=abc123...                                    │
│                                                                     │
│  5. Server ใช้ Session ID ดึงข้อมูลผู้ใช้                            │
│                                                                     │
└─────────────────────────────────────────────────────────────────────┘

    ตัวอย่าง¶

    ┌──────────┐                     ┌──────────┐
│ Browser  │  ─── Login ────►    │  Server  │
│          │  ◄── Session ID ──  │          │
│          │                     │          │
│          │  ─── Request ────►  │          │
│          │  (with Session ID)  │          │
│          │  ◄── "สวัสดี John" ──│          │
└──────────┘                     └──────────┘

    การใช้ Session ใน PHP¶

    เริ่มใช้ Session¶

    <?php
// ต้องเรียก session_start() ก่อนใช้งาน $_SESSION
session_start();

// เก็บข้อมูล
$_SESSION['user_id'] = 1;
$_SESSION['user_name'] = 'John';

// อ่านข้อมูล
echo $_SESSION['user_name']; // John

// ตรวจสอบว่ามีค่าหรือไม่
if (isset($_SESSION['user_id'])) {
    echo "Logged in!";
}

// ลบข้อมูล
unset($_SESSION['user_id']);

// ลบทั้งหมด (Logout)
session_destroy();
?>

    สร้าง auth-check.php¶

    ไฟล์นี้ใช้ include ใน API ที่ต้อง login

    โครงสร้างไฟล์¶

    backend-mysql/
├── config/
│   └── database.php
└── includes/
    └── auth-check.php    ← สร้างไฟล์นี้

    โค้ด¶

    <?php
/**
 * Auth Check
 * ตรวจสอบว่า user login แล้วหรือยัง
 *
 * วิธีใช้: require_once '../includes/auth-check.php';
 */

session_start();

// ตรวจสอบว่ามี session หรือไม่
if (!isset($_SESSION['user_id'])) {
    http_response_code(401);
    header('Content-Type: application/json; charset=utf-8');
    echo json_encode([
        'status' => 401,
        'message' => 'Unauthorized. Please login first.'
    ]);
    exit;
}

// ดึงข้อมูล user จาก session
$currentUserId = $_SESSION['user_id'];
$currentUserEmail = $_SESSION['user_email'];
$currentUserName = $_SESSION['user_name'];
$currentUserRole = $_SESSION['user_role'];

/**
 * ตรวจสอบว่าเป็น Admin หรือไม่
 */
function requireAdmin() {
    global $currentUserRole;

    if ($currentUserRole !== 'admin') {
        http_response_code(403);
        header('Content-Type: application/json; charset=utf-8');
        echo json_encode([
            'status' => 403,
            'message' => 'Forbidden. Admin access required.'
        ]);
        exit;
    }
}
?>

    วิธีใช้งาน¶

    <?php
// api/products/create.php - เฉพาะ Admin

require_once '../../config/database.php';
require_once '../../includes/auth-check.php';  // ต้อง login

requireAdmin();  // ต้องเป็น admin

// ถ้าผ่าน = เป็น admin แน่นอน
// ทำงานต่อ...
?>

    สร้าง logout.php¶

    โค้ด¶

    <?php
/**
 * Logout API
 * POST /api/auth/logout
 *
 * ออกจากระบบโดยลบ session
 */

require_once '../../config/database.php';

// ต้องเป็น POST
if ($_SERVER['REQUEST_METHOD'] !== 'POST') {
    sendResponse(405, 'Method not allowed. Use POST.');
}

// เริ่ม session
session_start();

// ลบข้อมูลใน session
$_SESSION = [];

// ลบ session cookie
if (isset($_COOKIE[session_name()])) {
    setcookie(session_name(), '', time() - 3600, '/');
}

// ทำลาย session
session_destroy();

sendResponse(200, 'Logout successful');
?>

    ทดสอบ¶

    • Method: POST
    • URL: http://localhost/vshop/backend-mysql/api/auth/logout.php

    Response:

    {
    "status": 200,
    "message": "Logout successful"
}

    สร้าง me.php¶

    API สำหรับดึงข้อมูล user ที่ login อยู่

    โค้ด¶

    <?php
/**
 * Get Current User API
 * GET /api/auth/me
 *
 * ดึงข้อมูล user ที่ login อยู่
 */

require_once '../../config/database.php';
require_once '../../includes/auth-check.php';

// ต้องเป็น GET
if ($_SERVER['REQUEST_METHOD'] !== 'GET') {
    sendResponse(405, 'Method not allowed. Use GET.');
}

// ดึงข้อมูล user จาก database (ข้อมูลล่าสุด)
try {
    $stmt = $pdo->prepare("
        SELECT id, email, name, role, created_at
        FROM users
        WHERE id = ?
    ");
    $stmt->execute([$currentUserId]);
    $user = $stmt->fetch();

    if (!$user) {
        // Session มี แต่ user ถูกลบไปแล้ว
        session_destroy();
        sendResponse(401, 'User not found. Session expired.');
    }

    sendResponse(200, 'User retrieved', [
        'user' => $user
    ]);

} catch (PDOException $e) {
    sendResponse(500, 'Database error: ' . $e->getMessage());
}
?>

    ทดสอบ¶

    • Method: GET
    • URL: http://localhost/vshop/backend-mysql/api/auth/me.php

    Response (Login แล้ว):

    {
    "status": 200,
    "message": "User retrieved",
    "data": {
        "user": {
            "id": 1,
            "email": "admin@vshop.com",
            "name": "Admin",
            "role": "admin",
            "created_at": "2024-01-01 00:00:00"
        }
    }
}

    Response (ยังไม่ Login):

    {
    "status": 401,
    "message": "Unauthorized. Please login first."
}

    สรุปไฟล์ Auth¶

    backend-mysql/
├── config/
│   └── database.php       # เชื่อมต่อ DB + helper functions
├── includes/
│   └── auth-check.php     # ตรวจสอบ login
└── api/
    └── auth/
        ├── login.php      # POST - เข้าสู่ระบบ
        ├── register.php   # POST - สมัครสมาชิก
        ├── logout.php     # POST - ออกจากระบบ
        └── me.php         # GET  - ดูข้อมูลตัวเอง

    Flow การใช้งาน¶

    ┌─────────────────────────────────────────────────────────────────────┐
│                        Authentication Flow                          │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  1. User สมัครสมาชิก                                                 │
│     POST /api/auth/register                                         │
│     { name, email, password }                                       │
│                                                                     │
│  2. User เข้าสู่ระบบ                                                 │
│     POST /api/auth/login                                            │
│     { email, password }                                             │
│     → Server สร้าง session                                          │
│                                                                     │
│  3. User ใช้งานระบบ                                                  │
│     GET /api/auth/me                                                │
│     GET /api/cart                                                   │
│     → ส่ง session cookie ไปด้วยทุกครั้ง                              │
│                                                                     │
│  4. User ออกจากระบบ                                                  │
│     POST /api/auth/logout                                           │
│     → Server ลบ session                                             │
│                                                                     │
└─────────────────────────────────────────────────────────────────────┘
