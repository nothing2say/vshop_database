    ดูประวัติการสั่งซื้อ (get-all.php)¶

    <?php
/**
 * Get Orders API
 * GET /api/orders/get-all.php
 *
 * ดูประวัติการสั่งซื้อของ user ที่ login
 */

require_once '../../config/database.php';
require_once '../../includes/auth-check.php';

if ($_SERVER['REQUEST_METHOD'] !== 'GET') {
    sendResponse(405, 'Method not allowed. Use GET.');
}

// Pagination
$page = max(1, (int)($_GET['page'] ?? 1));
$limit = min(50, max(1, (int)($_GET['limit'] ?? 10)));
$offset = ($page - 1) * $limit;

try {
    // นับจำนวนทั้งหมด
    $countStmt = $pdo->prepare("
        SELECT COUNT(*) as total
        FROM orders
        WHERE user_id = ?
    ");
    $countStmt->execute([$currentUserId]);
    $totalItems = (int) $countStmt->fetch()['total'];
    $totalPages = ceil($totalItems / $limit);

    // ดึง orders
    $stmt = $pdo->prepare("
        SELECT
            o.id,
            o.total_amount,
            o.status,
            o.created_at,
            (SELECT COUNT(*) FROM order_items WHERE order_id = o.id) AS item_count
        FROM orders o
        WHERE o.user_id = ?
        ORDER BY o.created_at DESC
        LIMIT ? OFFSET ?
    ");
    $stmt->execute([$currentUserId, $limit, $offset]);
    $orders = $stmt->fetchAll();

    sendResponse(200, 'Orders retrieved', [
        'orders' => $orders,
        'pagination' => [
            'current_page' => $page,
            'total_pages' => $totalPages,
            'total_items' => $totalItems
        ]
    ]);

} catch (PDOException $e) {
    sendResponse(500, 'Database error: ' . $e->getMessage());
}
?>

    ดูรายละเอียดคำสั่งซื้อ (get-by-id.php)¶

    <?php
/**
 * Get Order Details API
 * GET /api/orders/get-by-id.php?id=1
 *
 * ดูรายละเอียดคำสั่งซื้อพร้อมรายการสินค้า
 */

require_once '../../config/database.php';
require_once '../../includes/auth-check.php';

if ($_SERVER['REQUEST_METHOD'] !== 'GET') {
    sendResponse(405, 'Method not allowed. Use GET.');
}

$orderId = $_GET['id'] ?? null;

if (!$orderId) {
    sendResponse(400, 'Order ID is required');
}

try {
    // ===================================
    // ดึงข้อมูล order
    // ===================================

    $orderStmt = $pdo->prepare("
        SELECT
            id,
            user_id,
            total_amount,
            status,
            created_at
        FROM orders
        WHERE id = ? AND user_id = ?
    ");
    $orderStmt->execute([$orderId, $currentUserId]);
    $order = $orderStmt->fetch();

    if (!$order) {
        sendResponse(404, 'Order not found');
    }

    // ===================================
    // ดึงรายการสินค้าใน order
    // ===================================

    $itemsStmt = $pdo->prepare("
        SELECT
            oi.id,
            oi.product_id,
            oi.quantity,
            oi.price,
            (oi.price * oi.quantity) AS subtotal,
            p.name AS product_name,
            p.image_url
        FROM order_items oi
        JOIN products p ON oi.product_id = p.id
        WHERE oi.order_id = ?
    ");
    $itemsStmt->execute([$orderId]);
    $items = $itemsStmt->fetchAll();

    // ===================================
    // จัดรูปแบบ Response
    // ===================================

    sendResponse(200, 'Order retrieved', [
        'order' => [
            'id' => (int) $order['id'],
            'total_amount' => (float) $order['total_amount'],
            'status' => $order['status'],
            'created_at' => $order['created_at'],
            'items' => $items
        ]
    ]);

} catch (PDOException $e) {
    sendResponse(500, 'Database error: ' . $e->getMessage());
}
?>

    Admin: ดูทุก Orders¶

    <?php
/**
 * Admin: Get All Orders API
 * GET /api/admin/orders.php
 *
 * ดูคำสั่งซื้อทั้งหมดในระบบ (Admin only)
 */

require_once '../../config/database.php';
require_once '../../includes/auth-check.php';

requireAdmin();

if ($_SERVER['REQUEST_METHOD'] !== 'GET') {
    sendResponse(405, 'Method not allowed. Use GET.');
}

// Filters
$status = $_GET['status'] ?? null;
$page = max(1, (int)($_GET['page'] ?? 1));
$limit = min(50, max(1, (int)($_GET['limit'] ?? 20)));
$offset = ($page - 1) * $limit;

try {
    $sql = "
        SELECT
            o.id,
            o.total_amount,
            o.status,
            o.created_at,
            u.name AS customer_name,
            u.email AS customer_email,
            (SELECT COUNT(*) FROM order_items WHERE order_id = o.id) AS item_count
        FROM orders o
        JOIN users u ON o.user_id = u.id
    ";
    $params = [];

    if ($status) {
        $sql .= " WHERE o.status = ?";
        $params[] = $status;
    }

    $sql .= " ORDER BY o.created_at DESC LIMIT ? OFFSET ?";
    $params[] = $limit;
    $params[] = $offset;

    $stmt = $pdo->prepare($sql);
    $stmt->execute($params);
    $orders = $stmt->fetchAll();

    sendResponse(200, 'Orders retrieved', $orders);

} catch (PDOException $e) {
    sendResponse(500, 'Database error: ' . $e->getMessage());
}
?>

    Admin: อัพเดท Order Status¶

    <?php
/**
 * Admin: Update Order Status API
 * PUT /api/admin/orders/update-status.php?id=1
 *
 * Input: { status: "completed" | "cancelled" }
 */

require_once '../../../config/database.php';
require_once '../../../includes/auth-check.php';

requireAdmin();

if ($_SERVER['REQUEST_METHOD'] !== 'PUT') {
    sendResponse(405, 'Method not allowed. Use PUT.');
}

$orderId = $_GET['id'] ?? null;

if (!$orderId) {
    sendResponse(400, 'Order ID is required');
}

$input = getJSONInput();
$status = $input['status'] ?? null;

$validStatuses = ['pending', 'completed', 'cancelled'];
if (!in_array($status, $validStatuses)) {
    sendResponse(400, 'Invalid status. Use: ' . implode(', ', $validStatuses));
}

try {
    // ตรวจสอบ order
    $checkStmt = $pdo->prepare("SELECT id FROM orders WHERE id = ?");
    $checkStmt->execute([$orderId]);

    if (!$checkStmt->fetch()) {
        sendResponse(404, 'Order not found');
    }

    // Update status
    $updateStmt = $pdo->prepare("
        UPDATE orders
        SET status = ?, updated_at = NOW()
        WHERE id = ?
    ");
    $updateStmt->execute([$status, $orderId]);

    sendResponse(200, 'Order status updated', [
        'order_id' => (int) $orderId,
        'status' => $status
    ]);

} catch (PDOException $e) {
    sendResponse(500, 'Database error: ' . $e->getMessage());
}
?>
