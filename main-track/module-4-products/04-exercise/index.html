    แบบฝึกหัด Module 4¶

    แบบฝึกหัดที่ 1: SQL Query¶

    โจทย์ 1.1¶

    เขียน SQL ดึงสินค้าที่ราคามากกว่า 50,000 บาท เรียงตามราคาจากมากไปน้อย

    คำตอบ 
    SELECT id, name, price
FROM products
WHERE price > 50000
AND is_active = 1
ORDER BY price DESC;

    โจทย์ 1.2¶

    เขียน SQL ดึงสินค้าพร้อมชื่อหมวดหมู่ โดยแสดงเฉพาะหมวด "Notebooks"

    คำตอบ 
    SELECT
    p.id,
    p.name,
    p.price,
    c.name AS category_name
FROM products p
JOIN categories c ON p.category_id = c.id
WHERE c.name = 'Notebooks'
AND p.is_active = 1;

    โจทย์ 1.3¶

    เขียน SQL นับจำนวนสินค้าในแต่ละหมวดหมู่

    คำตอบ 
    SELECT
    c.name AS category_name,
    COUNT(p.id) AS product_count
FROM categories c
LEFT JOIN products p ON c.id = p.category_id AND p.is_active = 1
GROUP BY c.id
ORDER BY product_count DESC;

    แบบฝึกหัดที่ 2: Pagination¶

    โจทย์ 2.1¶

    คำนวณ OFFSET สำหรับ: - หน้า 5 - แสดง 20 รายการต่อหน้า

    คำตอบ 
    OFFSET = (page - 1) * limit
OFFSET = (5 - 1) * 20
OFFSET = 80

SQL: SELECT * FROM products LIMIT 20 OFFSET 80;

    โจทย์ 2.2¶

    มีสินค้าทั้งหมด 57 รายการ แสดง 10 รายการต่อหน้า มีกี่หน้า?

    คำตอบ 
    total_pages = ceil(total_items / limit)
total_pages = ceil(57 / 10)
total_pages = ceil(5.7)
total_pages = 6 หน้า

    แบบฝึกหัดที่ 3: API Search¶

    โจทย์ 3.1¶

    เพิ่ม feature ค้นหาสินค้าตามช่วงราคา

    Query params: - min_price: ราคาขั้นต่ำ - max_price: ราคาสูงสุด

    ตัวอย่าง: /api/products/get-all.php?min_price=10000&max_price=50000

    คำตอบ 
    <?php
// เพิ่มใน get-all.php

$minPrice = $_GET['min_price'] ?? null;
$maxPrice = $_GET['max_price'] ?? null;

// ... หลังจากสร้าง $sql และ $params ...

if ($minPrice) {
    $sql .= " AND p.price >= ?";
    $params[] = $minPrice;
}

if ($maxPrice) {
    $sql .= " AND p.price <= ?";
    $params[] = $maxPrice;
}
?>

    แบบฝึกหัดที่ 4: Featured Products¶

    โจทย์ 4.1¶

    สร้าง API get-featured.php ที่: - ดึงสินค้า 6 รายการล่าสุด - เรียงตาม created_at ใหม่สุดก่อน - แสดงเฉพาะที่ is_active = 1

    คำตอบ 
    <?php
// api/products/get-featured.php

require_once '../../config/database.php';

if ($_SERVER['REQUEST_METHOD'] !== 'GET') {
    sendResponse(405, 'Method not allowed. Use GET.');
}

try {
    $stmt = $pdo->query("
        SELECT
            p.id,
            p.name,
            p.price,
            p.image_url,
            c.name AS category_name
        FROM products p
        LEFT JOIN categories c ON p.category_id = c.id
        WHERE p.is_active = 1
        ORDER BY p.created_at DESC
        LIMIT 6
    ");
    $products = $stmt->fetchAll();

    sendResponse(200, 'Featured products retrieved', $products);

} catch (PDOException $e) {
    sendResponse(500, 'Database error: ' . $e->getMessage());
}
?>

    แบบฝึกหัดที่ 5: Products by Category¶

    โจทย์ 5.1¶

    สร้าง API get-by-category.php ที่: - รับ parameter category_id - ดึงสินค้าในหมวดหมู่นั้น - รองรับ pagination

    คำตอบ 
    <?php
// api/products/get-by-category.php

require_once '../../config/database.php';

if ($_SERVER['REQUEST_METHOD'] !== 'GET') {
    sendResponse(405, 'Method not allowed. Use GET.');
}

$categoryId = $_GET['category_id'] ?? null;
$page = max(1, (int)($_GET['page'] ?? 1));
$limit = min(50, max(1, (int)($_GET['limit'] ?? 10)));
$offset = ($page - 1) * $limit;

if (!$categoryId) {
    sendResponse(400, 'category_id is required');
}

try {
    // ตรวจสอบ category
    $catStmt = $pdo->prepare("SELECT id, name FROM categories WHERE id = ?");
    $catStmt->execute([$categoryId]);
    $category = $catStmt->fetch();

    if (!$category) {
        sendResponse(404, 'Category not found');
    }

    // นับจำนวน
    $countStmt = $pdo->prepare("
        SELECT COUNT(*) as total
        FROM products
        WHERE category_id = ? AND is_active = 1
    ");
    $countStmt->execute([$categoryId]);
    $totalItems = (int) $countStmt->fetch()['total'];
    $totalPages = ceil($totalItems / $limit);

    // ดึงสินค้า
    $stmt = $pdo->prepare("
        SELECT id, name, price, stock, image_url
        FROM products
        WHERE category_id = ? AND is_active = 1
        ORDER BY id DESC
        LIMIT ? OFFSET ?
    ");
    $stmt->execute([$categoryId, $limit, $offset]);
    $products = $stmt->fetchAll();

    sendResponse(200, 'Products retrieved', [
        'category' => $category,
        'products' => $products,
        'pagination' => [
            'current_page' => $page,
            'total_pages' => $totalPages,
            'total_items' => $totalItems
        ]
    ]);

} catch (PDOException $e) {
    sendResponse(500, 'Database error: ' . $e->getMessage());
}
?>
